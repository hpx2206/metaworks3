=== 적용 방법 ===
1. lib 에 있는 jar 들을 새로 생성하는 web 프로젝트에 넣어주어야함

2. 메타데이터 서비스를 사용하기 위하여, (image) web.xml 에 아래와 같은 코드를 넣어주어야함

<servlet>
    <description></description>
    <display-name>MetadataServlet</display-name>
    <servlet-name>MetadataServlet</servlet-name>
    <servlet-class>org.metaworks.metadata.MetadataServlet</servlet-class>
  </servlet>
  <servlet-mapping>
    <servlet-name>MetadataServlet</servlet-name>
    <url-pattern>/metadata/*</url-pattern>
  </servlet-mapping>
  
    <filter>
        <filter-name>TenantAwareFilter</filter-name>
        <filter-class>org.uengine.cloud.saasfier.TenantAwareFilter</filter-class>
        <init-param>
            <param-name>encoding</param-name>
            <param-value>UTF-8</param-value>
        </init-param>
    </filter>
    
    <filter-mapping>
        <filter-name>TenantAwareFilter</filter-name>
        <url-pattern>/*</url-pattern>
    </filter-mapping>
